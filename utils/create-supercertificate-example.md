 --manual-auth-hook ./cert-azure-auth.sh \
 --manual-cleanup-hook ./cert-azure-cleanup.sh \
  --preferred-challenges=dns \
  --email hostmaster@az.skillscloud.company \
  --server https://acme-v02.api.letsencrypt.org/directory \
